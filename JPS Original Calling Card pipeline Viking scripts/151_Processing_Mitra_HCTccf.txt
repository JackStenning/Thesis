#SBATCH --job-name=MitraCCF		 # Job name
#SBATCH --mail-type=END,FAIL             # Mail events (NONE, BEGIN, END, FAIL, ALL)
#SBATCH --mail-user=jps558@york.ac.uk    # Where to send mail  
#SBATCH --ntasks=1                       # Run on a single CPU
#SBATCH --mem=100gb                      # Job memory request
#SBATCH --time=12:00:00                  # Time limit hrs:min:sec
#SBATCH --output=basic_job_%j.log        # Standard output and error log
#SBATCH --account=biol-hic-2020          # Project account
module purge
module load bio/SAMtools/1.9-foss-2018b
module load bio/BEDTools/2.30.0-GCC-11.2.0
module load lang/Miniconda3
source activate Callingcardspy

cd workingCCdata/HCT-116
for file in *HyPBase.ccf.txt; do
	bedtools sort -i "$file" > "$file"_sorted.qbed
	python ../../SegmentCCF.py "$file"_sorted.qbed | sed -e '/^\s*$/d' > "$file"_sorted.qbed.blocks
	done
for dile in *HCT-116_HyPBase.*_sorted.qbed
	python ../../BBPeakCaller_BRD4.py -p 30 -d 12500 -i "$dile"_intermediate.csv \
	"$dile" \
	"$dile".blocks \
	../../staging/hg38/hg38_TTAA.bed \
	../../processedCCdata/20230713_ccfprocess/HCT-116/"$dile"_p30_peaks.bed
	done
for mile in *HCT-116_SP1-HyPBase*_sorted.qbed; do
	python ../../BBPeakCaller_TF.py -a 0.05 -m fdr_bh -d 250 -x 5000 \
	-i "$mile"_intermediate.csv \
	"$mile" \
	"$mile".blocks \
	"$dile" \
	../../processedCCdata/20230713_ccfprocess/HCT-116/"$mile"_fdr_bh_0.05_peaks.bed
	done
cd /mnt/lustre/groups/biol-hic-2020/jack/
