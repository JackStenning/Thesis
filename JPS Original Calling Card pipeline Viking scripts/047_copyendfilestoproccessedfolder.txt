#!/bin/bash

#Copies fastq, bw and bedgraph files from workingChipdata

for dir in workingChipdata/*; do
	cd /mnt/lustre/groups/biol-hic-2020/jack/"$dir"
	mkdir /mnt/lustre/groups/biol-hic-2020/jack/processedChipdata/pre230212/"$dir"
	cp *.bedgraph /mnt/lustre/groups/biol-hic-2020/jack/processedChipdata/pre230212/"$dir"
	cp *.bw /mnt/lustre/groups/biol-hic-2020/jack/processedChipdata/pre230212/"$dir"
	cp *.fastq /mnt/lustre/groups/biol-hic-2020/jack/processedChipdata/pre230212/"$dir"
	echo "Files from $dir were copied"
	cd ../..
	done
